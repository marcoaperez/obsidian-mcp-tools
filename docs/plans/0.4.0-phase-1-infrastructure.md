# 0.4.0 — Phase 1: Infrastructure foundation

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Land the in-process HTTP MCP server skeleton in the Obsidian plugin so that an empty-but-working 0.4.0-alpha.1 build answers `initialize`, `tools/list` (empty), and a single smoke-test tool (`get_server_info`) over the Streamable HTTP transport, with full auth and middleware in place. Tool porting and semantic search are out of scope for this phase and land in Phase 2 and Phase 3 respectively.

**Architecture:** The plugin instantiates a Node `http.Server` bound to `127.0.0.1:27200` (fallback 27201-27205), wraps it with a middleware chain (method/path → Origin validation → Bearer auth → StreamableHTTPServerTransport), and hands requests off to an `@modelcontextprotocol/sdk` `McpServer` that registers tools through a ported `ToolRegistry`. No Obsidian vault access in this phase — the only tool is a pure-info smoke test.

**Tech Stack:** TypeScript strict + `verbatimModuleSyntax`, Bun test runner, Node `http` built-in, `@modelcontextprotocol/sdk@1.29.0` (`StreamableHTTPServerTransport`, `McpServer`), ArkType 2.x for schema validation, Obsidian plugin runtime (Electron-embedded Node).

**Design reference:** `docs/design/2026-04-24-http-embedded-design.md`. Decisions referenced by ID (D1-D9) in-line.

**Branch:** `feat/http-embedded` (already created off `main`).

// token.ts
import { randomBytes, timingSafeEqual } from "node:crypto";
import { TOKEN_BYTE_LENGTH } from "../constants";

export function generateToken(): string {
  return randomBytes(TOKEN_BYTE_LENGTH).toString("base64url");
}

export function compareTokens(a: string, b: string): boolean {
  if (a.length !== b.length) return false;
  const aBuf = Buffer.from(a);
  const bBuf = Buffer.from(b);
  return timingSafeEqual(aBuf, bBuf);
}

// port.ts
import type { Server } from "node:http";
import { BIND_HOST } from "../constants";

export async function bindWithFallback(
  server: Server,
  ports: readonly number[],
): Promise<number> {
  for (const port of ports) {
    try {
      await new Promise<void>((resolve, reject) => {
        const onError = (err: NodeJS.ErrnoException) => {
          server.off("listening", onListening);
          reject(err);
        };
        const onListening = () => {
          server.off("error", onError);
          resolve();
        };
        server.once("error", onError);
        server.once("listening", onListening);
        server.listen(port, BIND_HOST);
      });
      return port;
    } catch (err) {
      if ((err as NodeJS.ErrnoException).code !== "EADDRINUSE") throw err;
      // Try next port
    }
  }
  throw new Error(`No free port in range: ${ports.join(", ")}`);
}
```

// middleware.ts
import { ERROR_CODES, MCP_PATH_PREFIX } from "../constants";

export type MethodPathResult =
  | { ok: true }
  | { ok: false; status: 404 | 405 };

const ALLOWED_METHODS = new Set(["GET", "POST"]);

export function checkMethodAndPath(
  method: string | undefined,
  url: string | undefined,
): MethodPathResult {
  const path = (url ?? "").split("?")[0];
  if (path !== MCP_PATH_PREFIX && !path.startsWith(`${MCP_PATH_PREFIX}/`)) {
    return { ok: false, status: ERROR_CODES.NOT_FOUND };
  }
  if (!ALLOWED_METHODS.has((method ?? "").toUpperCase())) {
    return { ok: false, status: ERROR_CODES.METHOD_NOT_ALLOWED };
  }
  return { ok: true };
}

import { compareTokens } from "./token";
import { isOriginAllowed } from "./origin";
import { ERROR_CODES } from "../constants";

export type RequestHeaders = Record<string, string | string[] | undefined>;

export type MiddlewareRequest = {
  method: string | undefined;
  url: string | undefined;
  headers: RequestHeaders;
};

export type MiddlewareResult =
  | { ok: true }
  | { ok: false; status: 401 | 403 | 404 | 405 };

function getHeader(headers: RequestHeaders, name: string): string | undefined {
  const v = headers[name.toLowerCase()];
  return Array.isArray(v) ? v[0] : v;
}

function checkAuth(headers: RequestHeaders, expectedToken: string): MiddlewareResult {
  const auth = getHeader(headers, "authorization");
  if (!auth) return { ok: false, status: ERROR_CODES.UNAUTHORIZED };
  const match = /^Bearer\s+(.+)$/.exec(auth);
  if (!match) return { ok: false, status: ERROR_CODES.UNAUTHORIZED };
  const token = match[1].trim();
  if (!compareTokens(token, expectedToken)) {
    return { ok: false, status: ERROR_CODES.UNAUTHORIZED };
  }
  return { ok: true };
}

function checkOrigin(headers: RequestHeaders): MiddlewareResult {
  const origin = getHeader(headers, "origin");
  return isOriginAllowed(origin)
    ? { ok: true }
    : { ok: false, status: ERROR_CODES.ORIGIN_FORBIDDEN };
}

export function runMiddleware(
  req: MiddlewareRequest,
  bearerToken: string,
): MiddlewareResult {
  const methodPath = checkMethodAndPath(req.method, req.url);
  if (!methodPath.ok) return methodPath;

  const origin = checkOrigin(req.headers);
  if (!origin.ok) return origin;

  const auth = checkAuth(req.headers, bearerToken);
  if (!auth.ok) return auth;

  return { ok: true };
}
```

// httpServer.ts
import { createServer, type IncomingMessage, type Server, type ServerResponse } from "node:http";
import { runMiddleware } from "./middleware";
import { bindWithFallback } from "./port";
import { PORT_RANGE } from "../constants";

export type RequestHandler = (req: IncomingMessage, res: ServerResponse) => Promise<void>;

export type HttpServerConfig = {
  bearerToken: string;
  requestHandler: RequestHandler;
};

export type RunningServer = {
  server: Server;
  port: number;
};

export async function startHttpServer(config: HttpServerConfig): Promise<RunningServer> {
  const server = createServer((req, res) => {
    const check = runMiddleware(
      { method: req.method, url: req.url, headers: req.headers },
      config.bearerToken,
    );
    if (!check.ok) {
      res.writeHead(check.status);
      res.end();
      return;
    }
    void config.requestHandler(req, res).catch((err) => {
      // Log and return 500 on unhandled handler error
      // (logger import deferred to keep this file dependency-light;
      //  wired via feature setup in Task 13.)
      if (!res.headersSent) res.writeHead(500);
      res.end();
      throw err;
    });
  });

  const port = await bindWithFallback(server, [...PORT_RANGE]);
  return { server, port };
}

export async function stopHttpServer({ server }: RunningServer): Promise<void> {
  await new Promise<void>((resolve, reject) => {
    server.close((err) => (err ? reject(err) : resolve()));
  });
}

// mcpServer.ts
import { McpServer } from "@modelcontextprotocol/sdk/server/mcp.js";
import { StreamableHTTPServerTransport } from "@modelcontextprotocol/sdk/server/streamableHttp.js";
import type { IncomingMessage, ServerResponse } from "node:http";
import { ToolRegistry } from "./toolRegistry";
import { registerTools } from "$/features/mcp-tools";

export type McpServiceConfig = {
  pluginVersion: string;
};

export type McpService = {
  server: McpServer;
  transport: StreamableHTTPServerTransport;
  registry: ToolRegistry;
  handleRequest: (req: IncomingMessage, res: ServerResponse) => Promise<void>;
};

export async function createMcpService(config: McpServiceConfig): Promise<McpService> {
  const server = new McpServer({
    name: "mcp-connector",
    version: config.pluginVersion,
  });

  const registry = new ToolRegistry(server);
  await registerTools(registry, { pluginVersion: config.pluginVersion });

  const transport = new StreamableHTTPServerTransport({
    sessionIdGenerator: undefined, // stateless mode: each request independent
  });

  await server.connect(transport);

  const handleRequest = async (req: IncomingMessage, res: ServerResponse) => {
    await transport.handleRequest(req, res);
  };

  return { server, transport, registry, handleRequest };
}

export async function destroyMcpService(svc: McpService): Promise<void> {
  await svc.transport.close();
  await svc.server.close();
}

// services/setup.ts
import type { Plugin } from "obsidian";
import { logger } from "$/shared";
import { startHttpServer, stopHttpServer, type RunningServer } from "./httpServer";
import { createMcpService, destroyMcpService, type McpService } from "./mcpServer";
import { generateToken } from "./token";

export type McpTransportState = {
  server: RunningServer;
  mcp: McpService;
  bearerToken: string;
};

export type SetupResult =
  | { success: true; state: McpTransportState }
  | { success: false; error: string };

export async function setup(plugin: Plugin): Promise<SetupResult> {
  try {
    const settings = (await plugin.loadData()) ?? {};
    let bearerToken = settings.mcpTransport?.bearerToken as string | undefined;
    if (!bearerToken || bearerToken.length < 32) {
      bearerToken = generateToken();
      await plugin.saveData({
        ...settings,
        mcpTransport: { ...(settings.mcpTransport ?? {}), bearerToken },
      });
    }

    const pluginVersion = plugin.manifest.version;
    const mcp = await createMcpService({ pluginVersion });
    const server = await startHttpServer({
      bearerToken,
      requestHandler: mcp.handleRequest,
    });

    logger.info("MCP Connector HTTP server listening", {
      port: server.port,
      pluginVersion,
    });

    return { success: true, state: { server, mcp, bearerToken } };
  } catch (err) {
    const message = err instanceof Error ? err.message : String(err);
    logger.error("MCP Connector failed to start HTTP server", { error: message });
    return { success: false, error: message };
  }
}

export async function teardown(state: McpTransportState): Promise<void> {
  await stopHttpServer(state.server);
  await destroyMcpService(state.mcp);
}
